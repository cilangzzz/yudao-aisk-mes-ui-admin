## 五、接口调用示例

### 5.1 API 函数示例

```typescript
// api/mes/work-order/index.ts
import request from '@/utils/request'
import type { MesWorkOrderPageReqVO, MesWorkOrderRespVO } from './types'

// 获取工单分页列表
export function getWorkOrderPage(params: MesWorkOrderPageReqVO) {
  return request.get<{ list: MesWorkOrderRespVO[], total: number }>('/mes/work-order/page', { params })
}

// 创建工单
export function createWorkOrder(data: MesWorkOrderSaveReqVO) {
  return request.post<Long>('/mes/work-order/create', data)
}

// 下发工单
export function releaseWorkOrder(id: number) {
  return request.put<boolean>('/mes/work-order/release', { id })
}
```

### 5.2 页面调用示例

```vue
<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { getWorkOrderPage, releaseWorkOrder } from '@/api/mes/work-order'
import type { MesWorkOrderRespVO } from '@/api/mes/work-order/types'

const tableData = ref<MesWorkOrderRespVO[]>([])
const loading = ref(false)

// 获取列表
const getList = async () => {
  loading.value = true
  try {
    const { list, total } = await getWorkOrderPage({ pageNo: 1, pageSize: 10 })
    tableData.value = list
  } finally {
    loading.value = false
  }
}

// 下发工单
const handleRelease = async (id: number) => {
  await releaseWorkOrder(id)
  ElMessage.success('下发成功')
  getList()
}

onMounted(() => {
  getList()
})
</script>
```
